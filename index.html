-----------------------------------------------------------
endpoint: /users
action : create new users
params : { username, email, password, blockchainWalletAddress, privateKey, score, enabled }
use :
curl -X POST http://localhost:8000/users -H "Content-Type: application/json" -d '{"username":"johndoe", "email":"john@example.com", "password":"123456", "blockchainWalletAddress":"0x123456A...", "privateKey":"myPrivateKeyA", "score":0, "enabled":true}'

-----------------------------------------------------------
endpoint: /users/user/:userId
action : get user by id
params : {userId}
use :
curl -X GET http://localhost:8000/users/user/{userId} -H "Content-Type: application/json"
note : returns [] if not found

-----------------------------------------------------------
endpoint: /login
action : login user based on credentials
params : { email, password }
use :
curl -X POST http://localhost:8000/login -H "Content-Type: application/json" -d '{"email":"john@example.com","password":"123456"}' note : returns "invalid password/email" "login successful"

-----------------------------------------------------------
endpoint: /accounts
action : create new bank account for userId
params : { userId, name, accountNumber, CBU, alias }
use :
curl -X POST http://localhost:8000/accounts -H "Content-Type: application/json" -d '{"user":"{userId}","name":"Main Account","accountNumber":"123456789","CBU":"1234567890123456789012","alias":"main_account"}'

-----------------------------------------------------------
endpoint: /accounts/user/:userId
action : Read all bank accounts for a user
params : {userId}
use :
curl -X GET http://localhost:8000/accounts/user/{userId} -H "Content-Type: application/json"

-----------------------------------------------------------
endpoint: /accounts/account/:accId
action : get bank account details
params : {accId}
use :
curl -X GET http://localhost:8000/accounts/account/{accId} -H "Content-Type: application/json"

-----------------------------------------------------------
endpoint: /wallets
action : add external wallet address
params : {USER_ID,WALLET_ADDRESS,BLOCKCHAIN_ID,WALLET_ALIAS}
use :
curl -X POST http://localhost:8000/wallets -H "Content-Type: application/json" -d '{"user":"66fb110945b92ad1013c16d6", "walletAddress":"0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045", "blockchainId":"1", "walletAlias":"Vitalik ETH"}'

-----------------------------------------------------------
endpoint: /wallets/user/:USER_ID
action : get all external wallets for user
params : {USER_ID}
use :
curl http://localhost:8000/wallets/user/66fb110945b92ad1013c16d6 -H "Content-Type: application/json"

-----------------------------------------------------------
endpoint: /coins
action : add coins/tokens for blockchain network ID
params : {contractAddress,blockchainId,symbol,name,decimals,image}
use :
curl -X POST http://localhost:8000/coins -H "Content-Type: application/json" -d '{"contractAddress":"0x86Fa049857E0209aa7D9e616F7eb3b3B78ECfdb0", "blockchainId":"1", "symbol":"EOS", "name":"EOS", "decimals":18, "image":""}'

-----------------------------------------------------------
endpoint: /coins/network/:NET_ID
action : get all tokens for blockchain network id
params : {NET_ID}
use :
curl http://localhost:8000/coins/network/1 -H "Content-Type: application/json"
